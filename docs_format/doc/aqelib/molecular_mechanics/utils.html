def parametrize_ligand(dockedmol: object,  # str | Chem.Mol
                       workdir: str) -> object:
    writer = Chem.SDWriter(os.path.join(workdir, "ligand.sdf"))
    if isinstance(dockedmol, str):
        dockedmol: Chem.Mol = Chem.JSONToMols(dockedmol)[0]
    writer.write(dockedmol)
    writer.flush()

    antechamber_cmd = ["antechamber", "-fi", "sdf", "-fo", "mol2",
                       "-i", os.path.join(workdir, "ligand.sdf"),
                       "-o", os.path.join(workdir, "ligand.mol2"),
                       "-nc", str(Chem.GetFormalCharge(dockedmol)),
                       "-c", "bcc"]
    logger.info(f"antechamber_cmd is {antechamber_cmd}")
    process = subprocess.Popen(
        antechamber_cmd,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        cwd=workdir)
    stdout, stderr = process.communicate()
    if stderr:
        process.kill()
        logger.error("""
        |- Cannot Antechamber : command error -|
        |- Antechamber: STDERR -|
        %s
        |- Antechamber: STDOUT -|
        %s
        """, bytes.decode(stderr, "ascii"), bytes.decode(stdout, "ascii"))
    process.terminate()

    parmchk2_cmd = ["parmchk2", "-f", "mol2",
                    "-i", os.path.join(workdir, "ligand.mol2"),
                    "-o", os.path.join(workdir, "ligand.frcmod")
                    ]
    logger.info(f"parmchk2_cmd is {parmchk2_cmd}")

    process = subprocess.Popen(
        parmchk2_cmd,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        cwd=workdir)

    stdout, stderr = process.communicate()
    if stderr:
        process.kill()
        logger.error("""
        |- Cannot Parmchk2 : command error -|
        |- Parmchk2: STDERR -|
        %s
        |- Parmchk2: STDOUT -|
        %s
        """, bytes.decode(stderr, "ascii"), bytes.decode(stdout, "ascii"))
    process.terminate()

    f = open(os.path.join(workdir, "tleap.in"), "w")
    f.write("""source leaprc.gaff
ligand = loadmol2 ligand.mol2
loadamberparams ligand.frcmod
saveamberparm ligand ligand.prmtop ligand.rst7
quit""")
    f.close()

    tleap_cmd = ["tleap", "-f", os.path.join(workdir, "tleap.in")]
    logger.info(f"tleap_cmd is {tleap_cmd}")
    process = subprocess.Popen(
        tleap_cmd,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        cwd=workdir)

    stdout, stderr = process.communicate()
    if stderr:
        process.kill()
        logger.error("""
        |- Cannot Tleap : command error -|
        |- Tleap: STDERR -|
        %s
        |- Tleap: STDOUT -|
        %s
        """, bytes.decode(stderr, "ascii"), bytes.decode(stdout, "ascii"))
    process.terminate()

    ligand = parmed.load_file(os.path.join(workdir, "ligand.prmtop"),
                              os.path.join(workdir, "ligand.rst7"))
    for res in ligand.residues:
        res.name = 'LIG'

    return ligand

def create_dict_per_atom(parmed_pose: str) -> list:
    """
    Create a dictionary for each atom in a molecule
    from a topology file and a coordinate file

    Parameters
    ----------

    topology_file : str
        Topology file of the molecule
    coordinate_file : str
        Coordinate file of the molecule
    parmed_pose : parmed object

    Returns
    -------

    atoms : list
    list of atom dictionaries containing :

        - charge
        - atomic_number
        - atom_reference
        - sigma
        - epsilon
        - x_coord
        - y_coord
        - z_coord

    """

    number_of_atoms = parmed_pose.ptr('NATOM')
    charges = parmed_pose.parm_data['CHARGE']
    atomic_numbers = parmed_pose.parm_data['ATOMIC_NUMBER']
    atom_reference = parmed_pose.parm_data['ATOM_NAME']

    kcal_to_kj = 4.184
    sigma_types = np.array(parmed_pose.LJ_radius) * 2.0 / \
        (2 ** (1 / 6))  # convert LJ radius to sigma
    epsilon_types = np.array(parmed_pose.LJ_depth) * \
        kcal_to_kj  # convert kcal/mol to kJ/mol

    sigma = []
    epsilon = []
    for idx in parmed_pose.parm_data['ATOM_TYPE_INDEX']:
        sigma.append(sigma_types[idx - 1])
        epsilon.append(epsilon_types[idx - 1])

    coordinates = parmed_pose.coordinates

    x_coords = coordinates[:, 0]
    y_coords = coordinates[:, 1]
    z_coords = coordinates[:, 2]

    atoms = []
    for i in range(number_of_atoms):
        atom_dict = {
            "charge": charges[i],
            "atomic_number": atomic_numbers[i],
            "atom_reference": atom_reference[i],
            "sigma": sigma[i],
            "epsilon": epsilon[i],
            "x_coord": x_coords[i],
            "y_coord": y_coords[i],
            "z_coord": z_coords[i]
        }
        atoms.append(atom_dict)

    return atoms


def calculate_sigma_ij(atoms_i, atoms_j, method='LB'):
    """
        Returns sigma_ij value by applying Lorentz-Berthelot
        (default) or geometric combining rule

    Parameters
    ----------

    atoms_i : dict
        dictionary containing properties of atom I
    atoms_j : dict
        dictionary containing properties of atom J
    method : str, optional
        Method name for calculating sigma_ij.
        Only "LB" and "geometric" are supported currently.
        The default is 'LB'.

    Returns
    -------

    sigma_ij : float
        value of sigma_ij

    """
    if method == 'LB':
        sigma_ij = (atoms_i['sigma'] + atoms_j['sigma']) / 2
    elif method == 'geometric':
        sigma_ij = np.sqrt(atoms_i['sigma'] * atoms_j['sigma'])
    else:
        raise RuntimeError(
            f"Invalid method for computing LJ sigma parameters: {method}")

    return sigma_ij


def calculate_epsilon_ij(atoms_i, atoms_j, method='LB'):
    """
    Returns epsilon_ij value by applying Lorentz-Berthelot
    (default) combining rule

    Parameters
    ----------

    atoms_a : dict
        dictionary containing properties of atom I
    atoms_b : dict
        dictionary containing properties of atom J
    method : str, optional
        Method name for calculating epsilon_ij.
        Only "LB" is supported currently.
        The default is 'LB'.

    Returns
    -------

    epsilon_ij : float
        value of epsilon_ij

    """
    if method == 'LB':
        epsilon_ij = np.sqrt(atoms_i['epsilon'] * atoms_j['epsilon'])
    else:
        raise RuntimeError(
            f"Invalid method for computing LJ sigma parameters: {method}")
    return epsilon_ij


def get_mol_coords(molecule):
    """
    Return list of atom coordinates:

    - one list for x_coords,
    - one y_coords,
    - one for z_coords

    Parameters
    ----------

    molecule : list
        list of atoms dictionary containing properties of atom i

    Returns
    -------
    x_coords : list
    y_coords : list
    z_coords : list

    """
    xyz_coords = np.array(
        [[atom["x_coord"], atom["y_coord"], atom["z_coord"]]
         for atom in molecule])

    return xyz_coords


def get_mol_charges(molecule: list) -> np.array:
    """
    Parameters
    ----------

    molecule : list
        list of atoms dictionary containing properties of atom i

    Returns
    -------

    charges: np.array
    """

    charges = np.array([atom["charge"] for atom in molecule])

    return charges


def get_mol_epsilon(molecule: list) -> np.array:
    """
    Parameters
    ----------

    molecule : list of atoms dictionary containing properties of atom i

    Returns
    -------

    epsilons
    """

    epsilons = np.array([atom["epsilon"] for atom in molecule])

    return epsilons


def get_mol_sigma(molecule):
    """
    Return list of atom sigmas

    Parameters
    ----------

    molecule : list
        list of atoms dictionary containing properties of atom i

    Returns
    -------

        sigmas : list
    """

    sigmas = np.array([atom["sigma"] for atom in molecule])

    return sigmas


def compute_distance_matrix(molecule_a, molecule_b):
    from scipy.spatial import distance_matrix
    """
    Generate a np.array
    containing the distance between each atoms of two molecules
    Distance is calculated with scipy.spatial.distance_matrix

    Parameters
    ----------

    molecule_a :list
        list of atoms dictionary containing properties of atom i
    molecule_b : list
        list of atoms dictionary containing properties of atom i

    Returns
    -------

    molecule_ab_distance_matrix : np.array
        shape : len(molecule_a) x len(molecule_b)
    """
    molecule_a_coords = get_mol_coords(molecule_a)
    molecule_b_coords = get_mol_coords(molecule_b)
    molecule_ab_distance_matrix = distance_matrix(
        molecule_a_coords, molecule_b_coords)

    return molecule_ab_distance_matrix


def compute_electrostatic_energy(molecule_a: list, molecule_b: list,
                                 relative_permittivity: float = 1) -> float:
    """
    Computes electrostatic energy between two molecules.

    Parameters
    ----------

    molecule_a : list
        list of atoms dictionaries containing properties of atom i.
    molecule_b : list
        list of atoms dictionaries containing properties of atom i.
    relative_permittivity : list
        The default is 1.

    Returns
    -------

    electrostatic_energy_kcal:
    the electrostatic energy between two molecules, in kcal/mol
    """
    n_avogadro = 6.02214076e23
    electron_volt = 1.60217662e-19
    epsilon0 = 8.85418782e-12
    angstrom_to_meter = 1.e-10
    kjoules_to_kcal = 0.239006

    permittivity = (relative_permittivity * epsilon0 *
                    angstrom_to_meter * 1000) / n_avogadro

    molecule_ab_distance_matrix = compute_distance_matrix(
        molecule_a, molecule_b)

    molecule_a_charges = get_mol_charges(molecule_a)
    molecule_b_charges = get_mol_charges(molecule_b)

    qiqj_matrix = (molecule_a_charges.reshape((len(molecule_a_charges), 1))
                   * molecule_b_charges.reshape((1, len(molecule_b_charges))))
    qiqj_matrix = qiqj_matrix * electron_volt * electron_volt

    electrostatic_energy = np.sum(
        qiqj_matrix / (molecule_ab_distance_matrix * 4 * np.pi * permittivity))

    electrostatic_energy_kcal = float(electrostatic_energy * kjoules_to_kcal)

    return electrostatic_energy_kcal


def compute_lennard_jones_energy(molecule_a: list, molecule_b: list) -> float:
    """
    Computes Lennard-Jones energy between two molecules.

    Parameters
    ----------

    molecule_a : list
        list of atoms dictionary containing properties of atom i.
    molecule_b : list
        list of atoms dictionary containing properties of atom i.

    Returns
    -------

    lennard_jones_energy_kcal:
    the Lennard-Jones energy between two molecules, in kcal/mol

    """
    kjoules_to_kcal = 0.239006

    molecule_ab_distance_matrix = compute_distance_matrix(
        molecule_a, molecule_b)

    molecule_a_epsilon = get_mol_epsilon(molecule_a)
    molecule_b_epsilon = get_mol_epsilon(molecule_b)
    epsilon_ij_matrix = np.sqrt(
        molecule_a_epsilon.reshape((len(molecule_a_epsilon), 1))
        * molecule_b_epsilon.reshape((1, len(molecule_b_epsilon))))

    molecule_a_sigma = get_mol_sigma(molecule_a)
    molecule_b_sigma = get_mol_sigma(molecule_b)
    sigma_ij_matrix = (molecule_a_sigma.reshape((len(molecule_a_sigma), 1))
                       + molecule_b_sigma.reshape((1, len(molecule_b_sigma)))
                       ) / 2

    sigma_distance_matrix_6 = (
        sigma_ij_matrix / molecule_ab_distance_matrix) ** 6
    lennard_jones_energy = np.sum(
        4 * epsilon_ij_matrix * sigma_distance_matrix_6 *
        (sigma_distance_matrix_6 - 1))

    lennard_jones_energy_kcal = float(lennard_jones_energy * kjoules_to_kcal)

    return lennard_jones_energy_kcal


def compute_interaction_energy(molecule_a: list,
                               molecule_b: list) -> (float, float, float):
    """
    Computes interaction energy between two molecules.

    Parameters
    ----------

    molecule_a : list
        list of atoms dictionary containing properties of atom i.
    molecule_b : list
        list of atoms dictionary containing properties of atom i.

    Returns
    -------

    interaction_energy, lennard_jones_energy, electrostatic_energy:
    the interaction energies between two molecules in kcal/mol
    """
    lennard_jones_energy = compute_lennard_jones_energy(molecule_a, molecule_b)
    electrostatic_energy = compute_electrostatic_energy(molecule_a, molecule_b)
    interaction_energy = lennard_jones_energy + electrostatic_energy

    return interaction_energy, lennard_jones_energy, electrostatic_energy


def compute_eint_from_top_crd(parmed_pose_l,
                              parmed_pose_p) -> (float, float, float):
    """
    Computes interaction energy between two molecules

    Parameters
    ----------
    parmed_pose_l : object
        parmed object
    parmed_pose_p : object
        parmed object

    Returns
    -------

    compute_interaction_energy: tuple
    The interaction energy between two molecules, in kcal/mol
    """
    ligand = create_dict_per_atom(parmed_pose_l)
    prot = create_dict_per_atom(parmed_pose_p)
    logger.info("Calculated Eint")
    e_int, e_lj, e_elec = compute_interaction_energy(ligand, prot)
    return e_int, e_lj, e_elec


def get_etot(system_parmed) -> float:
    import sander
    """
    get total energy with sander api

    Parameters
    ----------

    system_parmed : parmed Object

    Returns
    -------

    e_tot: float
    """
    system_parmed.save("temp.prmtop")

    sander_runner = sander
    opts = sander_runner.pme_input()
    opts.ntb = 1
    opts.ntc = 1
    opts.ntf = 1
    opts.cut = 8.0
    opts.igb = 0
    if sander_runner.is_setup():
        sander_runner.cleanup()
    with sander_runner.setup("temp.prmtop", system_parmed.coordinates,
                             [100, 100, 100, 90, 90, 90], opts):
        e, f = sander_runner.energy_forces()
    os.remove("temp.prmtop")
    return e.tot

    molecule_b : list
        list of atoms dictionary containing properties of atom i.

    Returns
    -------

    lennard_jones_energy_kcal:
    the Lennard-Jones energy between two molecules, in kcal/mol

    """
    kjoules_to_kcal = 0.239006

    molecule_ab_distance_matrix = compute_distance_matrix(
        molecule_a, molecule_b)

    molecule_a_epsilon = get_mol_epsilon(molecule_a)
    molecule_b_epsilon = get_mol_epsilon(molecule_b)
    epsilon_ij_matrix = np.sqrt(
        molecule_a_epsilon.reshape((len(molecule_a_epsilon), 1))
        * molecule_b_epsilon.reshape((1, len(molecule_b_epsilon))))

    molecule_a_sigma = get_mol_sigma(molecule_a)
    molecule_b_sigma = get_mol_sigma(molecule_b)
    sigma_ij_matrix = (molecule_a_sigma.reshape((len(molecule_a_sigma), 1))
                       + molecule_b_sigma.reshape((1, len(molecule_b_sigma)))
                       ) / 2

    sigma_distance_matrix_6 = (
        sigma_ij_matrix / molecule_ab_distance_matrix) ** 6
    lennard_jones_energy = np.sum(
        4 * epsilon_ij_matrix * sigma_distance_matrix_6 *
        (sigma_distance_matrix_6 - 1))

    lennard_jones_energy_kcal = float(lennard_jones_energy * kjoules_to_kcal)

    return lennard_jones_energy_kcal

def create_dict_per_atom(parmed_pose: str) -> list:
    """
    Create a dictionary for each atom in a molecule
    from a topology file and a coordinate file

    Parameters
    ----------

    topology_file : str
        Topology file of the molecule
    coordinate_file : str
        Coordinate file of the molecule
    parmed_pose : parmed object

    Returns
    -------

    atoms : list
    list of atom dictionaries containing :

        - charge
        - atomic_number
        - atom_reference
        - sigma
        - epsilon
        - x_coord
        - y_coord
        - z_coord

    """

    number_of_atoms = parmed_pose.ptr('NATOM')
    charges = parmed_pose.parm_data['CHARGE']
    atomic_numbers = parmed_pose.parm_data['ATOMIC_NUMBER']
    atom_reference = parmed_pose.parm_data['ATOM_NAME']

    kcal_to_kj = 4.184
    sigma_types = np.array(parmed_pose.LJ_radius) * 2.0 / \
        (2 ** (1 / 6))  # convert LJ radius to sigma
    epsilon_types = np.array(parmed_pose.LJ_depth) * \
        kcal_to_kj  # convert kcal/mol to kJ/mol

    sigma = []
    epsilon = []
    for idx in parmed_pose.parm_data['ATOM_TYPE_INDEX']:
        sigma.append(sigma_types[idx - 1])
        epsilon.append(epsilon_types[idx - 1])

    coordinates = parmed_pose.coordinates

    x_coords = coordinates[:, 0]
    y_coords = coordinates[:, 1]
    z_coords = coordinates[:, 2]

    atoms = []
    for i in range(number_of_atoms):
        atom_dict = {
            "charge": charges[i],
            "atomic_number": atomic_numbers[i],
            "atom_reference": atom_reference[i],
            "sigma": sigma[i],
            "epsilon": epsilon[i],
            "x_coord": x_coords[i],
            "y_coord": y_coords[i],
            "z_coord": z_coords[i]
        }
        atoms.append(atom_dict)

    return atoms

    import sander
    """
    get total energy with sander api

    Parameters
    ----------

    system_parmed : parmed Object

    Returns
    -------

    e_tot: float
    """
    system_parmed.save("temp.prmtop")

    sander_runner = sander
    opts = sander_runner.pme_input()
    opts.ntb = 1
    opts.ntc = 1
    opts.ntf = 1
    opts.cut = 8.0
    opts.igb = 0
    if sander_runner.is_setup():
        sander_runner.cleanup()
    with sander_runner.setup("temp.prmtop", system_parmed.coordinates,
                             [100, 100, 100, 90, 90, 90], opts):
        e, f = sander_runner.energy_forces()
    os.remove("temp.prmtop")
    return e.tot
