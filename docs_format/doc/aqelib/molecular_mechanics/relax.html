import pandas
import os
import shutil
import numpy
from rdkit import Chem
from .run_gromacs import relax_with_gromacs
from .utils import (compute_eint_from_top_crd, get_etot,
                    parametrize_ligand, parametrize_protein)
from ..mdft import write_affinity_input, write_affinity_parameters_file
from ..moltools import hash_mol
from ..loggers import chemlog

logger = chemlog.getLogger(__name__)


def relax_posesdf(poses_df: pandas.DataFrame,
                  meta: dict = {}) -> pandas.DataFrame:
    relaxed_posesdf = pandas.DataFrame(
        numpy.empty(0, dtype=list(meta.items())))
    for index, row in poses_df.iterrows():
        if row.protein_coord_hash == "foo":
            pass  # only usefull with dask
        workdir_protein = os.path.join("/tmp", row.protein_id,
                                       row.protein_coord_hash,
                                       f"{row.protein_id}.pdb")
        if not os.path.exists(workdir_protein):
            os.makedirs(os.path.dirname(workdir_protein), exist_ok=True)
            with open(workdir_protein, 'w') as file:
                file.write(row.protein_pdb)
                file.close()
        values = relax_pose(row.mol, row.protein_id, row.protein_coord_hash)
        if values:
            relaxed_posesdf = relaxed_posesdf.append(values, ignore_index=True)
    return relaxed_posesdf


def relax_pose(mol: str,  # str | Chem.Mol
               protein_id: str,
               protein_coord_hash: str) -> dict:
    workdir_protein = os.path.join("/tmp", protein_id, protein_coord_hash)
    hashmol = hash_mol(mol, str)
    if isinstance(mol, str):
        mol: Chem.Mol = Chem.JSONToMols(mol)[0]
    workdir_mol = os.path.join("/tmp", hashmol)
    logger.info(f"Will relax mol {hashmol} on protein {protein_coord_hash}")
    protein_pdb_path = os.path.join(workdir_protein,
                                    f"{protein_id}.pdb")
    if not os.path.exists(protein_pdb_path):
        raise FileNotFoundError(f"{protein_pdb_path} do not exist")
    try:
        shutil.rmtree(workdir_mol, ignore_errors=True)
        os.makedirs(workdir_mol, exist_ok=True)
        ligand_init = parametrize_ligand(
            mol, workdir=workdir_mol)
        logger.info("ligand_init DONE")
    except Exception as exception:
        logger.error("ligand_init was catch")
        logger.exception(exception)
        return None

    try:
        protein_init = parametrize_protein(
            protein_pdb_path, workdir=workdir_protein)
        logger.info("protein_init DONE")
    except Exception as exception:
        logger.error("protein_init was catch")
        logger.exception(exception)
        return None

    complex_init = protein_init + ligand_init
    relax_complex_dir = os.path.join(workdir_protein, "relax_complex")
    shutil.rmtree(relax_complex_dir, ignore_errors=True)
    os.makedirs(relax_complex_dir)
    try:
        (complex_relaxed,
         relaxation_steps,
         relaxation_time) = relax_with_gromacs(
            run_dir=relax_complex_dir,
            system_pmd=complex_init,
            total_minimization_cycles=100000)
    except Exception as exception:
        logger.error("relax_with_gromacs complex_relax was catch")
        logger.exception(exception)
        return None

    ligand_relaxed_in_protein = complex_relaxed[":LIG"]
    protein_relaxed = complex_relaxed["!(:LIG)"]

    e_inter, e_lj, e_elec = compute_eint_from_top_crd(
        parmed_pose_l=ligand_relaxed_in_protein,
        parmed_pose_p=protein_relaxed)

    e_tot_complex = get_etot(complex_relaxed)
    e_tot_protein = get_etot(protein_relaxed)
    e_tot_ligand_relaxed_in_protein = get_etot(ligand_relaxed_in_protein)
    complex_relaxed.save(os.path.join(
        workdir_protein, "complex_relaxed.pdb"), overwrite=True)
    complex_init.save(os.path.join(
        workdir_protein, "complex.prmtop"), overwrite=True)
    complex_relaxed.save(os.path.join(
        workdir_protein, "complex_relaxed.rst7"), overwrite=True)

    complex_in = write_affinity_input(complex_relaxed)
    complex_dft_in = write_affinity_parameters_file(
        system_in=complex_in,
        mdft_dxdydz=0.3)
    protein_in = write_affinity_input(protein_relaxed)
    protein_dft_in = write_affinity_parameters_file(
        system_in=protein_in,
        mdft_dxdydz=0.3)
    ligand_in = write_affinity_input(ligand_relaxed_in_protein)
    ligand_dft_in = write_affinity_parameters_file(
        system_in=ligand_in,
        mdft_dxdydz=0.3)

    relax_ligand_dir = os.path.join(workdir_mol, "relax_ligand")
    shutil.rmtree(relax_ligand_dir, ignore_errors=True)
    os.makedirs(relax_ligand_dir)
    try:
        (ligand_relaxed_in_vacuum,
         ligand_relaxation_steps,
         ligand_relaxation_time) = relax_with_gromacs(
            run_dir=relax_ligand_dir,
            system_pmd=ligand_relaxed_in_protein,
            total_minimization_cycles=100000)
    except Exception as exception:
        logger.error("relax_with_gromacs relax_ligand was catch")
        logger.exception(exception)
        return None

    e_tot_ligand_relaxed_in_vacuum = get_etot(ligand_relaxed_in_vacuum)

    ligand_relaxed_in_vacuum.save(os.path.join(
        workdir_mol, "ligand_relaxed_in_vacuum.rst7"), overwrite=True)

    with open(os.path.join(workdir_protein,
                           "complex_relaxed.pdb"), 'r') as file:
        complex_relaxed_pdb: str = file.read()
        file.close()
    with open(os.path.join(workdir_protein,
                           "complex.prmtop"), 'r') as file:
        complex_prmtop: str = file.read()
        file.close()
    with open(os.path.join(workdir_protein,
                           "complex_relaxed.rst7"), 'r') as file:
        complex_relaxed_rst7: str = file.read()
        file.close()
    with open(os.path.join(workdir_mol,
                           "ligand_relaxed_in_vacuum.rst7"), 'r') as file:
        ligand_relaxed_in_vacuum_rst7: str = file.read()
        file.close()

    data = {'inchikey': Chem.MolToInchiKey(mol, options="/FixedH"),
            'e_inter': e_inter,
            'e_lj': e_lj,
            'e_elec': e_elec,
            'e_tot_complex': e_tot_complex,
            'e_tot_protein': e_tot_protein,
            'e_tot_ligand_relaxed_in_protein': e_tot_ligand_relaxed_in_protein,
            'e_tot_ligand_relaxed_in_vacuum': e_tot_ligand_relaxed_in_vacuum,
            'relaxation_time': relaxation_time,
            'relaxation_steps': relaxation_steps,
            'complex_relaxed_pdb': complex_relaxed_pdb,
            'complex_prmtop': complex_prmtop,
            'complex_relaxed_rst7': complex_relaxed_rst7,
            'ligand_relaxed_in_vacuum_rst7': ligand_relaxed_in_vacuum_rst7,
            'complex_in': complex_in,
            'complex_dft_in': complex_dft_in,
            'protein_in': protein_in,
            'protein_dft_in': protein_dft_in,
            'ligand_in': ligand_in,
            'ligand_dft_in': ligand_dft_in,
            'protein_id': protein_id,
            'protein_coord_hash': protein_coord_hash,
            'pose_hash': hashmol}
    shutil.rmtree(workdir_protein, ignore_errors=True)
    return data
