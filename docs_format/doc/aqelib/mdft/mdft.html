import s3fs
import zipfile
import json
import os
import pandas


def zipdir(path, ziph):
    # ziph is zipfile handle
    for root, dirs, files in os.walk(path):
        for file in files:
            ziph.write(os.path.join(root, file), os.path.relpath(
                os.path.join(root, file), os.path.join(path, '..')))


def run_mdft_PL_P_L(relax_df: pandas.DataFrame,
                    aws_credentials: dict) \
        -> pandas.DataFrame:

    mdft_run_ids = []
    for index, relax_row in relax_df.iterrows():
        pose_hash = relax_row["pose_hash"]
        mdft_run_id_pl = run_mdft(solute_in=relax_row["complex_in"],
                                  solute_dft_in=relax_row["complex_dft_in"],
                                  system_name="PL",
                                  pose_hash=pose_hash,
                                  aws_credentials=aws_credentials)
        mdft_run_id_p = run_mdft(solute_in=relax_row["protein_in"],
                                 solute_dft_in=relax_row["protein_dft_in"],
                                 system_name="P",
                                 pose_hash=pose_hash,
                                 aws_credentials=aws_credentials)
        mdft_run_id_l = run_mdft(solute_in=relax_row["ligand_in"],
                                 solute_dft_in=relax_row["ligand_dft_in"],
                                 system_name="L",
                                 pose_hash=pose_hash,
                                 aws_credentials=aws_credentials)
        mdft_run_ids.append({
            "pose_hash": relax_row["pose_hash"],
            "mdft_run_id_pl": mdft_run_id_pl,
            "mdft_run_id_p": mdft_run_id_p,
            "mdft_run_id_l": mdft_run_id_l
        })
    return mdft_run_ids


def run_mdft(solute_in: str, solute_dft_in: str, pose_hash: str,
             system_name: str, aws_credentials: dict,
             workdir: str = "/tmp") -> float:
    mdft_run_id = f"{pose_hash}_{system_name}"
    in_root = os.path.join(workdir, mdft_run_id)
    if not os.path.exists(in_root):
        os.mkdir(in_root)

    base_dir = os.path.join(in_root, "input")
    if not os.path.exists(base_dir):
        os.mkdir(base_dir)

    input_dir = os.path.join(base_dir, "mdft_input")
    if not os.path.exists(input_dir):
        os.mkdir(input_dir)

    solute_in_path = os.path.join(input_dir, "solute.in")
    solute_dft_in_path = os.path.join(input_dir, "dft.in")
    config_json_path = os.path.join(in_root, "config.json")

    config_json = {
        "aws_email": "theau.debroise@aqemia.com",
        "in_root": "",
        "keep_all": True,
        "keep_dirs": "yes",
        "mdft_job_queue": "cCinqDeuxXlarge",
        "mdft_timeout": 1800,
        "mdft_token": "W2wHC7Uxahy3y7mpc",
        "mdft_username": "theau",
        "run_id": base_dir,
        "to_keep": [
            "mdft.out"
        ]
    }

    with open(config_json_path, 'w') as file:
        json.dump(config_json, file,
                  indent=4,
                  sort_keys=True)
        file.close()

    with open(solute_in_path, 'w') as file:
        file.write(solute_in)
        file.close()

    with open(solute_dft_in_path, 'w') as file:
        file.write(solute_dft_in)
        file.close()

    zipf = zipfile.ZipFile(
        os.path.join(in_root, "input.zip"), 'w', zipfile.ZIP_DEFLATED)
    zipdir(base_dir, zipf)
    zipf.close()

    ExtraArgs = {
        'GrantFullControl':
        'id="f6508f0d9faf02bf598dcaaee6753fc07891a1814e61700ab2caa170866c9ad7"'
    }

    s3 = s3fs.S3FileSystem(key=aws_credentials["AWS_ACCESS_KEY_ID"],
                           secret=aws_credentials["AWS_SECRET_ACCESS_KEY"],
                           s3_additional_kwargs=ExtraArgs)

    if not s3.exists(f"s3://mdftgate/aqelib_input/{mdft_run_id}/config.json"):
        s3_path = f"s3://mdftgate/aqelib_input/{mdft_run_id}/config.json"
        s3.put(config_json_path, s3_path)
    s3_path = f"s3://mdftgate/aqelib_input/{mdft_run_id}/input.zip"
    s3.put(os.path.join(in_root, "input.zip"), s3_path)
    return mdft_run_id


def get_mdft_dg_sol(mdft_run_id: str,
                    aws_credentials: dict,
                    workdir: str = "/tmp"):
    mdft_run_id_json = os.path.join(workdir, f"{mdft_run_id}.json")
    s3 = s3fs.S3FileSystem(key=aws_credentials["AWS_ACCESS_KEY_ID"],
                           secret=aws_credentials["AWS_SECRET_ACCESS_KEY"])
    s3_path = f"s3://mdftgate/aqelib_output/{mdft_run_id}.json"
    s3.get(s3_path, mdft_run_id_json)
    with open(mdft_run_id_json, 'r') as f:
        mdft_run_id_dict = json.load(f)
    return mdft_run_id_dict["dg_solv"]
